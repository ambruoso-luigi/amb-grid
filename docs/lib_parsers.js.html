const DEFAULT_DATE_OPTIONS = {
    inputFormat: 'dd/mm/yyyy',
    outputFormat: 'yyyy-mm-dd',
    allowEmpty: true
};

const DEFAULT_DATE_INPUT_FORMATS = [
    'dd/mm/yyyy',
    'dd-mm-yyyy',
    'dd.mm.yyyy',
    'mm/dd/yyyy',
    'mm-dd-yyyy',
    'yyyy-mm-dd',
    'yyyy/mm/dd',
    'yyyymmdd',
    Date
];

const isEmptyValue = value => {
    return value === null
        || value === undefined
        || (typeof value === 'string' && value.trim() === '');
};

const getEmptyResult = options => {
    return options.allowEmpty ? options.emptyAs : null;
};

const stringifyValue = value => {
    if (value === null || value === undefined) return '';

    return String(value);
};

const padDatePart = value => {
    return String(value).padStart(2, '0');
};

const normalizeDateFormat = format => {
    const aliases = {
        iso: 'yyyy-mm-dd',
        ISO: 'yyyy-mm-dd',
        it: 'dd/mm/yyyy',
        IT: 'dd/mm/yyyy',
        legacy: 'yyyymmdd'
    };

    return aliases[format] || format;
};

const isValidDateParts = (year, month, day) => {
    const date = new Date(year, month - 1, day);

    return date.getFullYear() === year
        && date.getMonth() === month - 1
        && date.getDate() === day;
};

const parseDateValue = (value, inputFormat) => {
    const normalizedInputFormat = normalizeDateFormat(inputFormat);

    if (normalizedInputFormat === Date || normalizedInputFormat === 'Date') {
        if (!(value instanceof Date) || !Number.isFinite(value.getTime())) return null;

        return {
            year: value.getFullYear(),
            month: value.getMonth() + 1,
            day: value.getDate()
        };
    }

    const stringValue = String(value).trim();
    let match;
    let year;
    let month;
    let day;

    if (normalizedInputFormat === 'dd/mm/yyyy') {
        match = /^(\d{1,2})\/(\d{1,2})\/(\d{4})$/.exec(stringValue);
        if (!match) return null;

        day = Number(match[1]);
        month = Number(match[2]);
        year = Number(match[3]);
    } else if (normalizedInputFormat === 'dd-mm-yyyy') {
        match = /^(\d{1,2})-(\d{1,2})-(\d{4})$/.exec(stringValue);
        if (!match) return null;

        day = Number(match[1]);
        month = Number(match[2]);
        year = Number(match[3]);
    } else if (normalizedInputFormat === 'dd.mm.yyyy') {
        match = /^(\d{1,2})\.(\d{1,2})\.(\d{4})$/.exec(stringValue);
        if (!match) return null;

        day = Number(match[1]);
        month = Number(match[2]);
        year = Number(match[3]);
    } else if (normalizedInputFormat === 'mm/dd/yyyy') {
        match = /^(\d{1,2})\/(\d{1,2})\/(\d{4})$/.exec(stringValue);
        if (!match) return null;

        month = Number(match[1]);
        day = Number(match[2]);
        year = Number(match[3]);
    } else if (normalizedInputFormat === 'mm-dd-yyyy') {
        match = /^(\d{1,2})-(\d{1,2})-(\d{4})$/.exec(stringValue);
        if (!match) return null;

        month = Number(match[1]);
        day = Number(match[2]);
        year = Number(match[3]);
    } else if (normalizedInputFormat === 'yyyy-mm-dd') {
        match = /^(\d{4})-(\d{1,2})-(\d{1,2})$/.exec(stringValue);
        if (!match) return null;

        year = Number(match[1]);
        month = Number(match[2]);
        day = Number(match[3]);
    } else if (normalizedInputFormat === 'yyyy/mm/dd') {
        match = /^(\d{4})\/(\d{1,2})\/(\d{1,2})$/.exec(stringValue);
        if (!match) return null;

        year = Number(match[1]);
        month = Number(match[2]);
        day = Number(match[3]);
    } else if (normalizedInputFormat === 'yyyymmdd') {
        match = /^(\d{4})(\d{2})(\d{2})$/.exec(stringValue);
        if (!match) return null;

        year = Number(match[1]);
        month = Number(match[2]);
        day = Number(match[3]);
    } else {
        return null;
    }

    if (!isValidDateParts(year, month, day)) return null;

    return { year, month, day };
};

const formatDateValue = ({ year, month, day }, outputFormat) => {
    const normalizedOutputFormat = normalizeDateFormat(outputFormat);
    const formattedMonth = padDatePart(month);
    const formattedDay = padDatePart(day);

    if (normalizedOutputFormat === 'dd/mm/yyyy') {
        return `${formattedDay}/${formattedMonth}/${year}`;
    }

    if (normalizedOutputFormat === 'dd-mm-yyyy') {
        return `${formattedDay}-${formattedMonth}-${year}`;
    }

    if (normalizedOutputFormat === 'dd.mm.yyyy') {
        return `${formattedDay}.${formattedMonth}.${year}`;
    }

    if (normalizedOutputFormat === 'mm/dd/yyyy') {
        return `${formattedMonth}/${formattedDay}/${year}`;
    }

    if (normalizedOutputFormat === 'mm-dd-yyyy') {
        return `${formattedMonth}-${formattedDay}-${year}`;
    }

    if (normalizedOutputFormat === 'yyyy-mm-dd') {
        return `${year}-${formattedMonth}-${formattedDay}`;
    }

    if (normalizedOutputFormat === 'yyyy/mm/dd') {
        return `${year}/${formattedMonth}/${formattedDay}`;
    }

    if (normalizedOutputFormat === 'yyyymmdd') {
        return `${year}${formattedMonth}${formattedDay}`;
    }

    if (normalizedOutputFormat === 'Date') {
        return new Date(year, month - 1, day);
    }

    return null;
};

const parseDateUsingFormats = (value, inputFormats) => {
    for (const inputFormat of inputFormats) {
        const dateParts = parseDateValue(value, inputFormat);

        if (dateParts) return dateParts;
    }

    return null;
};

const formatPayloadDate = dateParts => {
    return formatDateValue(dateParts, 'yyyy-mm-dd');
};

const escapeRegExp = value => {
    return String(value).replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
};

const normalizeNumericSign = (value, allowNegative) => {
    if (value.startsWith('-')) {
        if (!allowNegative) return null;

        return {
            sign: '-',
            unsigned: value.slice(1)
        };
    }

    if (value.startsWith('+')) {
        return null;
    }

    return {
        sign: '',
        unsigned: value
    };
};

const hasValidThousands = (value, thousandSeparator) => {
    const separator = escapeRegExp(thousandSeparator);
    const pattern = new RegExp(`^\\d{1,3}(?:${separator}\\d{3})+$`);

    return pattern.test(value);
};

const normalizeIntegerPart = (value, options) => {
    if (value === '') return null;

    if (value.includes(options.thousandSeparator)) {
        if (!options.allowThousands) return null;
        if (options.strictThousands && !hasValidThousands(value, options.thousandSeparator)) return null;
        if (!options.strictThousands && value.split(options.thousandSeparator).some(part => part === '')) return null;

        const normalized = value.split(options.thousandSeparator).join('');

        return /^\d+$/.test(normalized) ? normalized : null;
    }

    return /^\d+$/.test(value) ? value : null;
};

/**
 * Parser factories used to normalize values before payload/backend submission.
 *
 * Formatters display values, editors let users modify values, validators check
 * whether values are acceptable, and parsers transform values into predictable
 * backend-oriented shapes. Parsers may perform small syntactic checks to avoid
 * incoherent transformations, but they do not replace validators, business
 * rules, permissions, or backend validation.
 *
 * @namespace parsers
 */
export const parsers = {
    /**
     * Create a strict date parser.
     *
     * Supports `dd/mm/yyyy`, `dd-mm-yyyy`, `dd.mm.yyyy`, `mm/dd/yyyy`,
     * `mm-dd-yyyy`, `yyyy-mm-dd`, `yyyy/mm/dd`, `yyyymmdd`, `Date`, and the
     * `it`, `iso`, and `legacy` aliases. Separated formats accept one or two
     * digits for day/month; compact `yyyymmdd` stays strictly eight digits.
     * Invalid dates return `null`; empty values follow `allowEmpty`.
     *
     * @param {object} [options] - Date parser options.
     * @param {'dd/mm/yyyy'|'dd-mm-yyyy'|'dd.mm.yyyy'|'mm/dd/yyyy'|'mm-dd-yyyy'|'yyyy-mm-dd'|'yyyy/mm/dd'|'yyyymmdd'|'it'|'iso'|'legacy'|Date|string} [options.inputFormat='dd/mm/yyyy'] - Expected input format.
     * @param {'dd/mm/yyyy'|'dd-mm-yyyy'|'dd.mm.yyyy'|'mm/dd/yyyy'|'mm-dd-yyyy'|'yyyy-mm-dd'|'yyyy/mm/dd'|'yyyymmdd'|'it'|'iso'|'legacy'|'Date'} [options.outputFormat='yyyy-mm-dd'] - Returned format.
     * @param {boolean} [options.allowEmpty=true] - Whether empty values parse as an empty string.
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    date(options = {}) {
        const normalizedOptions = {
            ...DEFAULT_DATE_OPTIONS,
            emptyAs: '',
            ...options
        };

        return {
            parse(value) {
                if (isEmptyValue(value)) {
                    return getEmptyResult(normalizedOptions);
                }

                const dateParts = parseDateValue(value, normalizedOptions.inputFormat);

                if (!dateParts) return null;

                return formatDateValue(dateParts, normalizedOptions.outputFormat);
            }
        };
    },

    /**
     * Normalize a visual decimal value to a payload-oriented decimal string.
     *
     * The parser returns strings instead of JavaScript numbers to avoid
     * precision surprises with decimal and monetary values. It performs
     * syntactic checks only and does not replace decimal validators or backend
     * validation.
     *
     * @param {object} [options] - Decimal normalization options.
     * @param {string} [options.decimalSeparator=','] - Visual decimal separator.
     * @param {string} [options.thousandSeparator='.'] - Visual thousands separator.
     * @param {boolean} [options.allowThousands=true] - Allow grouped thousands.
     * @param {boolean} [options.allowNegative=true] - Allow negative values.
     * @param {string} [options.outputFormat='yyyy-mm-dd'] - Returned payload date format.
     * @param {boolean} [options.allowEmpty=true] - Allow empty values.
     * @param {''|null} [options.emptyAs=''] - Returned value for empty input.
     * @param {boolean} [options.strictThousands=true] - Require coherent thousands grouping.
     * @param {boolean} [options.allowNormalized=true] - Accept already normalized `123.45` values.
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    decimalToPayload(options = {}) {
        const normalizedOptions = {
            decimalSeparator: ',',
            thousandSeparator: '.',
            allowThousands: true,
            allowNegative: true,
            allowEmpty: true,
            emptyAs: '',
            strictThousands: true,
            allowNormalized: true,
            ...options
        };

        return {
            parse(value) {
                if (isEmptyValue(value)) return getEmptyResult(normalizedOptions);

                const signedValue = normalizeNumericSign(String(value).trim(), normalizedOptions.allowNegative);

                if (!signedValue || signedValue.unsigned === '') return null;

                if (
                    normalizedOptions.allowNormalized
                    && normalizedOptions.decimalSeparator !== '.'
                    && !signedValue.unsigned.includes(normalizedOptions.decimalSeparator)
                    && /^\d+\.\d+$/.test(signedValue.unsigned)
                    && !hasValidThousands(signedValue.unsigned, normalizedOptions.thousandSeparator)
                ) {
                    return `${signedValue.sign}${signedValue.unsigned}`;
                }

                const parts = signedValue.unsigned.split(normalizedOptions.decimalSeparator);

                if (parts.length > 2) return null;

                const integerPart = normalizeIntegerPart(parts[0], normalizedOptions);

                if (integerPart === null) return null;

                if (parts.length === 1) {
                    return `${signedValue.sign}${integerPart}`;
                }

                const decimalPart = parts[1];

                if (!/^\d+$/.test(decimalPart)) return null;

                return `${signedValue.sign}${integerPart}.${decimalPart}`;
            }
        };
    },

    /**
     * Normalize a visual integer value to a payload-oriented integer string.
     *
     * Do not use this parser for codes with leading zeroes. Codes are strings
     * and should be normalized with string normalizers instead.
     *
     * @param {object} [options] - Integer normalization options.
     * @param {string} [options.thousandSeparator='.'] - Visual thousands separator.
     * @param {boolean} [options.allowThousands=true] - Allow grouped thousands.
     * @param {boolean} [options.allowNegative=true] - Allow negative values.
     * @param {boolean} [options.allowEmpty=true] - Allow empty values.
     * @param {''|null} [options.emptyAs=''] - Returned value for empty input.
     * @param {boolean} [options.strictThousands=true] - Require coherent thousands grouping.
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    integerToPayload(options = {}) {
        const normalizedOptions = {
            thousandSeparator: '.',
            allowThousands: true,
            allowNegative: true,
            allowEmpty: true,
            emptyAs: '',
            strictThousands: true,
            ...options
        };

        return {
            parse(value) {
                if (isEmptyValue(value)) return getEmptyResult(normalizedOptions);

                const signedValue = normalizeNumericSign(String(value).trim(), normalizedOptions.allowNegative);

                if (!signedValue || signedValue.unsigned === '') return null;

                const integerPart = normalizeIntegerPart(signedValue.unsigned, normalizedOptions);

                return integerPart === null ? null : `${signedValue.sign}${integerPart}`;
            }
        };
    },

    /**
     * Normalize supported AMB Grid date formats to canonical payload `YYYY-MM-DD`.
     *
     * Ambiguous values are parsed using the configured `inputFormats` order.
     * Parsers do not guess user intent; configure `inputFormats` explicitly
     * when a data source can be ambiguous.
     *
     * @param {object} [options] - Date payload normalization options.
     * @param {Array<string|Date>} [options.inputFormats] - Accepted input formats, in priority order.
     * @param {boolean} [options.allowEmpty=true] - Allow empty values.
     * @param {''|null} [options.emptyAs=''] - Returned value for empty input.
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    dateToPayload(options = {}) {
        const normalizedOptions = {
            inputFormats: DEFAULT_DATE_INPUT_FORMATS,
            outputFormat: 'yyyy-mm-dd',
            allowEmpty: true,
            emptyAs: '',
            ...options
        };

        return {
            parse(value) {
                if (isEmptyValue(value)) return getEmptyResult(normalizedOptions);

                const dateParts = parseDateUsingFormats(value, normalizedOptions.inputFormats);

                return dateParts
                    ? formatDateValue(dateParts, normalizedOptions.outputFormat)
                    : null;
            }
        };
    },

    /**
     * Normalize supported AMB Grid date formats plus time to canonical payload `YYYY-MM-DD HH:MM:SS`.
     *
     * Timezone and Unix timestamp parsing are intentionally out of scope for
     * this parser. Timestamp-oriented parsers may be added later.
     *
     * @param {object} [options] - DateTime payload normalization options.
     * @param {Array<string|Date>} [options.inputFormats] - Accepted date input formats, in priority order.
     * @param {boolean} [options.allowEmpty=true] - Allow empty values.
     * @param {''|null} [options.emptyAs=''] - Returned value for empty input.
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    dateTimeToPayload(options = {}) {
        const normalizedOptions = {
            inputFormats: DEFAULT_DATE_INPUT_FORMATS,
            allowEmpty: true,
            emptyAs: '',
            ...options
        };

        return {
            parse(value) {
                if (isEmptyValue(value)) return getEmptyResult(normalizedOptions);

                const match = /^(.+)\s+(\d{2}):(\d{2})(?::(\d{2}))?$/.exec(String(value).trim());

                if (!match) return null;

                const dateParts = parseDateUsingFormats(match[1], normalizedOptions.inputFormats);

                if (!dateParts) return null;

                const hour = Number(match[2]);
                const minute = Number(match[3]);
                const second = match[4] === undefined ? 0 : Number(match[4]);

                if (hour < 0 || hour > 23) return null;
                if (minute < 0 || minute > 59) return null;
                if (second < 0 || second > 59) return null;

                return `${formatPayloadDate(dateParts)} ${padDatePart(hour)}:${padDatePart(minute)}:${padDatePart(second)}`;
            }
        };
    },

    /**
     * Trim leading and trailing whitespace.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    trim() {
        return {
            parse(value) {
                return stringifyValue(value).trim();
            }
        };
    },

    /**
     * Convert empty or whitespace-only values to `null`.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    emptyToNull() {
        return {
            parse(value) {
                return isEmptyValue(value) ? null : value;
            }
        };
    },

    /**
     * Convert text to uppercase.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    uppercase() {
        return {
            parse(value) {
                return stringifyValue(value).toUpperCase();
            }
        };
    },

    /**
     * Remove whitespace characters.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    removeSpaces() {
        return {
            parse(value) {
                return stringifyValue(value).replace(/\s+/g, '');
            }
        };
    },

    /**
     * Keep only decimal digit characters.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    digitsOnly() {
        return {
            parse(value) {
                return stringifyValue(value).replace(/\D+/g, '');
            }
        };
    },

    /**
     * Normalize IBAN text for payload/backend submission.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    ibanToPayload() {
        return {
            parse(value) {
                return stringifyValue(value).trim().replace(/\s+/g, '').toUpperCase();
            }
        };
    },

    /**
     * Normalize Italian Codice Fiscale text for payload/backend submission.
     *
     * @returns {{parse: Function}} Parser object with a `parse(value)` method.
     */
    fiscalCodeToPayload() {
        return {
            parse(value) {
                return stringifyValue(value).trim().replace(/\s+/g, '').toUpperCase();
            }
        };
    }
};
